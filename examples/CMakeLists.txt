cmake_minimum_required(VERSION 3.20)

project(BabyBehaveExamples VERSION 1.0)

set(CMAKE_CXX_STANDARD 23)
set(CMAKE_CXX_STANDARD_REQUIRED True)

add_executable(example_AlarmClock AlarmClock.cpp)
target_link_libraries(example_AlarmClock PRIVATE BabyBehave::BabyBehave babybehave_coverage_flags babybehave_sanitizer_flags)

add_executable(example_Calculator Calculator.cpp)
target_link_libraries(example_Calculator PRIVATE BabyBehave::BabyBehave babybehave_coverage_flags babybehave_sanitizer_flags)

add_executable(example_Failing Failing.cpp)
target_link_libraries(example_Failing PRIVATE BabyBehave::BabyBehave babybehave_coverage_flags babybehave_sanitizer_flags)

add_executable(example_FlightBooking FlightBooking.cpp)
target_link_libraries(example_FlightBooking PRIVATE BabyBehave::BabyBehave babybehave_coverage_flags babybehave_sanitizer_flags)

add_executable(example_Oven Oven.cpp)
target_link_libraries(example_Oven PRIVATE BabyBehave::BabyBehave babybehave_coverage_flags babybehave_sanitizer_flags)

add_executable(example_NoShortMacros NoShortMacros.cpp)
target_link_libraries(example_NoShortMacros PRIVATE BabyBehave::BabyBehave babybehave_coverage_flags babybehave_sanitizer_flags)

add_executable(example_Matchers Matchers.cpp)
target_link_libraries(example_Matchers PRIVATE BabyBehave::BabyBehave babybehave_coverage_flags babybehave_sanitizer_flags)

find_package(Threads REQUIRED)
add_executable(example_MultiThreaded MultiThreaded.cpp)
target_link_libraries(example_MultiThreaded PRIVATE BabyBehave::BabyBehave babybehave_coverage_flags babybehave_sanitizer_flags Threads::Threads)
